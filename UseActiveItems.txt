int * shuffleList(int *begin, int *end, RNG *rng){
	int len = end - begin
	for(int i=len-1;i>0;i--){
		int target = rng:NextInt(i)
		int temp = begin[i]
		begin[i] = begin[target]
		begin[target] = temp
	}
	return begin
}
